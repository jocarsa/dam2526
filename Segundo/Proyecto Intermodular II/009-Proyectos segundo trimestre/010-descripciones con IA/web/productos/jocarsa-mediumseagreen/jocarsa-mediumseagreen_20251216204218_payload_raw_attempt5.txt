     def summarize_transcript(transcript_path):
         with open(transcript_path, 'r', encoding='utf-8') as file:
             transcript = file.read()
         
         sentences = sent_tokenize(transcript)
         num_sentences = len(sentences)
         summary_length = int(num_sentences * 0.2)  # Generate a 20% summary
         
         frequency_distribution = nltk.FreqDist([word.lower() for sentence in sentences for word in sentence.split()])
         sorted_freq_dist = sorted(frequency_distribution.items(), key=lambda x: x[1], reverse=True)
         top_words = set(word for (word, _) in sorted_freq_dist[:50])  # Top 50 words
         
         summary = [sentence for sentence in sentences if any(word.lower() in sentence.lower() for word in top_words)]
         return ' '.join(sentence + '.' for sentence in summary)
