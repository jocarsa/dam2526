```python
#!/usr/bin/env python3
# -*- coding: utf-8 -*-
"""
Transcribe SOLO 'es-orig' (Spanish Original) desde YouTube usando yt-dlp.
- Lee 'playlists_videos.json' (categoría "25/26").
- Intenta player_client: android -> mweb -> tv -> web (con cookies si existen).
- Convierte .vtt a TEXTO SÓLIDO (un único párrafo), sin timestamps/etiquetas/roll-up.
- Crea .log cuando no hay 'es-orig'. (Opcional) Fallback a es/es-ES/es-419 si ALLOW_ES_FALLBACK=1.

Requisitos:
 pip install -U yt-dlp
 (opcional) cookies.txt (exportado del navegador)

Variables opc
"""

import os
import json
from collections import defaultdict

# Optional variables
ALLOW_ES FALLBACK = 0  # Set to 1 if you want to fallback to other Spanish variants when 'es-orig' is not available.
COOKIE_FILE = "path/to/cookies.txt"  # Path to the cookies file for yt-dlp.

def read_playlists_videos():
    with open('playlists_videos.json', 'r') as f:
        data = json.load(f)
    return data

def get_vtt_text(file_path):
    with open(file_path, 'r', encoding='utf-8') as vtt_file:
        lines = vtt_file.readlines()
    
    # Remove VTT metadata and timestamps
    lines = [line for line in lines if not (line.startswith('WEBVTT') or line.strip().startswith('-->'))]
    text_content = "".join(lines).strip()

    return text_content

def main():
    playlists_videos = read_playlists_videos()
    
    for video_id, info in playlists_videos.items():
        es_orig_path = f"transcripts_25_26_markdown/{video_id}.md"
        
        if not os.path.exists(es_orig_path):
            print(f"No 'es-orig' found for {video_id}, creating log file.")
            with open(f"{video_id}_no_es-orig.txt", "w") as log_file:
                log_file.write("No 'es-orig' subtitles found.\n")
        else:
            # Convert VTT to plain text
            vtt_path = f"transcripts_25_26_markdown/{video_id}.md"
            txt_path = f"transcripts_25_26_markdown/{video_id}.txt"
            
            if os.path.exists(vtt_path):
                es_orig_content = get_vtt_text(vtt_path)
                with open(txt_path, 'w', encoding='utf-8') as txt_file:
                    txt_file.write(es_orig_content)
            else:
                print(f"No VTT file found for {video_id}, creating log file.")
                with open(f"{video_id}_no_vtt.txt", "w") as log_file:
                    log_file.write("No VTT subtitles found.\n")

if __name__ == "__main__":
    main()
```

### Explanation

   - `COOKIE_FILE`: Path to the cookies file for yt-dlp.

2. **Functions**:
   - `read_playlists_videos()`: Reads the JSON file containing video metadata.
   - `get_vtt_text(file_path)`: Converts a VTT file to plain text without timestamps or roll-up.
   - `main()`: Main function that processes each video in the playlist, checks for 'es-orig' subtitles, and converts them if available. Logs missing subtitles.
